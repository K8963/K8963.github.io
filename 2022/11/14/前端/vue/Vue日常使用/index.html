父组件数据联动子组件

父组件向子组件传递数据，当父组件数据修改时，子组件一并修改，使用watch

// 子组件中
const props: any = defineProps({
	id: {
		type: Array,
		required: true
	},
  // 需要同步的数据
	eventData: {
		type: Array
	},
})

const eventTableData = ref<Event.Event[]>(props.eventData)
watch(
	() => props.eventData,
	() => {
		eventTableData.value = props.eventData
	},
	{ deep: true }
)
