JavaScript之this关键字

为什么要使用this

减少代码量，使代码整洁

未使用this

let myNewCar = {
  type:'truck',
  size:'big',
  description:fuunction(){
  	console.log("这个车的类型是：" + myNewCar.type + ',它的尺寸是：' + myNewCar.size);
	}
}
myNewCar.description()

使用this后

let myNewCar = {
  type:'truck',
  size:'big',
  description:fuunction(){
  	console.log("这个车的类型是：" + this.type + ',它的尺寸是：' + this.size);
	}
}
myNewCar.description()

this的绑定规则

默认绑定

function girl(){
  console.log(this)
}
girl()

此处的this默认绑定全局对象window

隐式绑定

let girl1 = {
  name: '小红',
  height: 160,
  weight: 180,
  datail: function () {
    console.log("小明的女朋友叫" + this.name + '，她身高' + this.height + '体重' + this.weight);
  }
}
girl1.datail();

此处的this隐式绑定对象girl1

硬绑定

console.log("---------硬绑定--------");
let girlName = {
  name: '小红',
  sayName: function () {
    console.log("我的女朋友叫" + this.name);
  }
}
let girl2 = {
  name: '小黄'
}
let girl3 = {
  name: '小绿'
}
girlName.sayName.call(girl2)
girlName.sayName.apply(girl3)

使用 call() 和 apply() 方法改变了this的指向

构造函数绑定

function Lover(name) {
  this.name = name;
  this.sayName = function () {
    console.log('小明的老婆是' + this.name);
  }
}
let name = '小黄';
let wife = new Lover('小红');
wife.sayName();

创建一个构造函数，然后对其实例化，这时候的this就会和实例化后的新对象进行捆绑，即使出现了同名的变量name，也不会影响name

题目测试

第一道

function a() {
  function b() {
    console.log(this);
    function c() {
      'use strict';
      console.log(this);
    }
    c();
  }
  b();
}
a();

第3行的this，没有指定任何对象，因此使用默认绑定指向全局window对象。

第6行的this，使用了严格模式，因此this为undefined

第二道

let name = '小绿';

function special() {
  console.log(this.name);
}
let girl = {
  name: '小红',
  datail: function () {
    console.log(this.name);
  },
  woman: {
    name: '小黄',
    datail: function () {
      console.log(this.name);
    }
  },
  special: special,
}
girl.datail();
girl.woman.datail();
girl.special();

19行的datail()方法是在girl对象中调用，因此this指向girl对象中的name

20行的datail()方法是在girl内部对象woman中调用，此处的datail()方法是woman中的datail()方法，因此this指向woman对象中的name

21行的special()方法虽然是全局方法，但是是在girl对象中调用，this会指向实际调用该方法的对象，因此this指向girl对象中的name
