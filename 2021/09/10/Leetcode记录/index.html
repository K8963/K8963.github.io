let s = "(])";
var isValid = function (s) {
  if (s.length % 2 === 1) {
    return false;
  }
  let stack = [];
  for (let i = 0; i < s.length; i++) {
    let ch = s.charAt(i);
    if (ch == "(" || ch == "[" || ch == "{") stack.push(ch);
    if (!stack.length) return false;
    if (ch == ")" && stack.pop() !== "(") return false;
    if (ch == "]" && stack.pop() !== "[") return false;
    if (ch == "}" && stack.pop() !== "{") return false;
  }
  return stack.length === 0;
};

  const n = s.length;

    return false;
  }
  const pairs = new Map([
    [")", "("],
    ["]", "["],
    ["}", "{"],
  ]);
  const stk = [];
  for (let ch of s) {
    if (pairs.has(ch)) {
      if (!stk.length || stk[stk.length - 1] !== pairs.get(ch)) {
        return false;
      }
      stk.pop();
    } else {
      stk.push(ch);
    }
  }
  return !stk.length;
};
isValid(s);
